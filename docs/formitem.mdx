## Examples

### Primary

```SnackPlayer name=Button&dependencies=fundamental-react-native@0.0.3
import React from 'react';
import { View } from 'react-native';
import { FormInput, FormItem, FormLabel, ThemeProvider, theme } from 'fundamental-react-native';

const App = () => (
  <ThemeProvider value={theme}>
    <View style={{
      flex: 1,
      justifyContent: 'center',
      alignItems: 'center',
    }}>
      <FormItem>
        <FormLabel>Default Input</FormLabel>
        <FormInput placeholder="Field placeholder text" />
      </FormItem>
    </View>
  </ThemeProvider>
);

export default App;
```
