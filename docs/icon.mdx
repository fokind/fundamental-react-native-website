## Examples

### Primary

```SnackPlayer name=Button&dependencies=fundamental-react-native@0.0.3,@expo/vector-icons
import React from 'react';
import { View } from 'react-native';
import { Icon, ThemeProvider, theme } from 'fundamental-react-native';

const App = () => (
  <ThemeProvider value={theme}>
    <View style={{
      flex: 1,
      justifyContent: 'center',
      alignItems: 'center',
    }}>
      <Icon iconSet="FontAwesome" glyph="shopping-cart" />
    </View>
  </ThemeProvider>
);

export default App;
```
